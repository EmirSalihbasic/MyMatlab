% If Statement

MyInput = input ('lets add a number: ');

if MyInput < 0 
     disp("you are negative!")   % disp is for display
else if (MyInput >= 0) && (MyInput <= 100)
     disp("you are right one")
else
     disp("Wow, thats a big number")
end    
end

% Switch statement

n = input ('Enter a number: ');
m = input ('Enter another number: ');
action = input ('What calculation should i perform: ');

switch action
       case 'add'
          output = n + m
       case 'subtract'
          output = n - m
       case 'multiply' 
          output = n * m
       case 'divide'
          output = n / m
       otherwise      % default case in other languages
          output ("sry, I can't do that.") 
end

Don't forget to add single quotes with the "operation"
